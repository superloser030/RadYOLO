\chapter{전체 디렉토리 트리}
\begin{lstlisting}[language=]
RadYOLO/
  main_r.py                 # receiver main: state machine + fusion + viewer
  src/
    transmission/
      sender.py             # TX: level select, DCA CLI, radar/webcam send
      receiver.py           # RX: chunk reassembly, MATLAB forward, meta gate
    objects/
      radar_fusion.py       # projection/match/aggregate (radar_to_x, match_all)
      obj_crop.py           # SAM2 fine segmentation + cutout
      trellis_gen.py        # TRELLIS (fal-ai) 3D generation
      pose_estimator.py     # GigaPose subprocess (incomplete)
    background/
      depth.py              # DA3 metric depth (live/bg/both)
      dynamic_bg_fill.py    # dynamic background filling
      upscale.py            # Real-ESRGAN upscale
      bg_select.py          # background frame selection (sharpness+SNR)
      depth_calibration.py  # (legacy) log-linear depth calibration
    utils/
      config.py             # TOML loader + level resolve
      gpu_scheduler.py      # priority GPU job queue / VRAM mgmt
      radar_cam_calib.py    # external calib (yaw) estimation loop
      archive.py            # archive previous data/
    viewer/
      viewer.html           # Three.js 2D/3D unified viewer + console
  matlab/
    radar_live.m            # realtime polling + CFAR pipeline
    modules/                # rangeDopplerMap, cfarDetect, estimateAzimuth ...
  workflows_comfyui/        # da3_depth.json (DA3), ESRGAN workflow
  config/
    network.toml            # IP/port
    sender.toml             # level/camera/DCA/radar params
    receiver.toml           # storage/pipeline/YOLO/DynBG
  tools/
    dca1000/                # TI DCA1000 capture CLI + chirp_configs
    gigapose/               # GigaPose (external repo, gitignore)
  data/                     # generated (gitignore)
    scene/                  # background, depth_metric, depth_bg, live_overlay
    objects/                # per-object views/cutout/glb/templates/pose
    radar/                  # targets.json, iqData_*.bin, .mat
    webcam/                 # received frames
  db/                       # object model DB (re-id, gitignore)
\end{lstlisting}

\chapter{실행 모드와 CLI 플래그}
\begin{center}
\begin{tabular}{@{}lp{8.5cm}@{}}
\toprule
플래그 & 의미 \\
\midrule
(없음)          & 전체 파이프라인: 배경촬영→ESRGAN→DA3→마스크→라이브 \\
\texttt{--calib} & 외부 캘리브(yaw) 모드(경량). 배경/3D 미생성 \\
\texttt{--verify}& 경량 수신 검증 모드 \\
\texttt{--skip-calib} & 레이더 기반 깊이 보정(Step 3.5) 생략(metric 모드에선 무의미) \\
\texttt{--skip-depth} & DA3 추론 생략(이전 depth 재사용) \\
\texttt{--no-radar}   & 레이더 없이 웹캠만(메타 대기 생략) \\
\bottomrule
\end{tabular}
\end{center}
